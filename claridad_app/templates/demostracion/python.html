Python + Tkinter


import tkinter as tk

class ClarityApp:
    def __init__(self, root):
        self.root = root
        self.root.title("Claridad Técnica")
        
        tk.Label(root, text="Ejemplo incorrecto:").pack()
        tk.Label(root, text="Apriete el tornillo. Verifique la pieza.", fg="red").pack()
        tk.Label(root, text="Ejemplo correcto:").pack()
        tk.Label(root, text="Apriete el tornillo. Verifique el tornillo.", fg="green").pack()

if __name__ == "__main__":
    root = tk.Tk()
    app = ClarityApp(root)
    root.mainloop()
